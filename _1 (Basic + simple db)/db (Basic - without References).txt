Create table admin(
		id serial Primary Key NOT NULl,
		login varchar(30),
		password varchar(30)
);

Create table booking(
		id serial Primary Key NOT NULl,
		bookingName varchar(100),
		cities varchar(50),
		seats varchar(100),
		price integer
);

Create table bus(
		id serial Primary Key NOT NULl,
		number integer,
		driverId integer,
		driverName varchar(100)
);

Create table driver(
		id serial Primary Key NOT NULl,
		driverId integer,
		password varchar(100),
		driverName varchar(100),
		busNumber integer
);

Create table payment(
		id serial Primary Key NOT NULl,
		cardName varchar(100),
		cardNumbers varchar(6),
		cvc_cvv integer
);

Create table schedule (
		id serial Primary Key NOT NULl,
		cities varchar(100),
		busNumber integer,
		timeStart varchar(5),
		timeEnd varchar(5)
);
